{% codeblock lang:sql %}
SELECT obj.Name SPName, sc.TEXT SPText
FROM sys.syscomments sc
INNER JOIN sys.objects obj ON sc.Id = obj.OBJECT_ID
WHERE sc.TEXT LIKE ‘%’ + ‘컬럼명’ + ‘%’
AND TYPE = ‘P’
{% endcodeblock %}
