```
docker run \
    -d \
    --name jenkins \
    --restart=always \
    -p 8081:8080 \
    -p 50000:50000 \
    -v /etc/localtime:/etc/localtime:ro \
    -v /var/jenkins_home:/var/jenkins_home \
    -u 0 \
    -e TZ=Asia/Seoul \
    jenkins
```
